
for n in $(ls ${workDir}/mapping/*/*filtered.fastq.gz)
do
    bn=$(basename $n _filtered.fastq.gz)
    cd ${workDir}/postprocessing/${bn}

    ## Allele-specific expression analysis
    $samtools idxstats ${bn}.realigned.bam > ${bn}.allreads.expression.txt

    ## Compute expression
    python ${scriptDir}/compute_expression.py ${bn}.allreads.expression.txt ${bn}.ASE.csv ${genomeDir}/${tRNAName}_clusterInfo.fa ${genomeDir}/tRNAs.txt ${bn}.RPM.allreads.csv

    ## Stats
    $samtools stats ${bn}.realigned.bam | grep ^RL | cut -f 2- > ${bn}.readlengths.allreads.txt
    $samtools depth ${bn}.realigned.bam > ${bn}.depth.allreads.txt

done
